```javascript
/**
 * @param {number} num
 * @return {string}
 */
var intToRoman = function(num) {
    let floor = Math.floor
    let res=  ""
    let cout = 0
    // 有值的情况
    if(cout = floor(num / 1000)) {
        while(cout) {
            res += "M"
            cout--
        }
        num %= 1000
    }

    if(cout = floor(num / 100)) {
        if(cout === 9) {
            res += "CM"
        } else if(cout === 4) {
            res += "CD"
        } else {
            if(cout >= 5) {
                res += "D"
                cout-=5
            } 
            while(cout) {
                res += "C"
                cout--
            }
        }
        num %= 100
    }


    if(cout = floor(num / 10)) {
        if(cout === 9) {
            res += "XC"
        } else if(cout === 4) {
            res += "XL"
        } else {
            if(cout >= 5) {
                res += "L"
                cout-=5
            } 
            while(cout) {
                res += "X"
                cout--
            }
        }
        num %= 10
    }


if(cout = floor(num / 1)) {
        if(cout === 9) {
            res += "IX"
        } else if(cout === 4) {
            res += "IV"
        } else {
            if(cout >= 5) {
                res += "V"
                cout-=5
            } 
            while(cout) {
                res += "I"
                cout--
            }
        }
    }


    
    return res;

};
```

这个是在题解中找到的一个比较简介的解法, 但感觉这种用减的方式会不会效率低? 要减好多次? 但我上面的循环好像也要循环多次


```javascript
var intToRoman = function (num) {
    const valueSymbols = [[1000, "M"], [900, "CM"], [500, "D"], [400, "CD"], [100, "C"], [90, "XC"], [50, "L"], [40, "XL"], [10, "X"], [9, "IX"], [5, "V"], [4, "IV"], [1, "I"]];
    const roman = [];
    for(const [value,symbol] of valueSymbols) {
        while(num >= value) {
            num -= value;
            roman.push(symbol)
        }
        if(num===0) {
            break
        }
    }
    return roman.join('')
}
```

::: tab Java解法
```java
import java.util.*;


public class Main {
  public static void main(String[] args) {
    Solution solution = new Solution();

    System.out.println(solution.intToRoman(19494589));
  }

}



class Solution {
  int[] values = {1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1};
  String[] symbols = {"M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I"};

  public String intToRoman(int num) {
    StringBuffer roman = new StringBuffer();
    for (int i = 0; i < values.length; i++) {
      int value = values[i];
      String symbol = symbols[i];
      while (num >= value) {
        num -= value;
        roman.append(symbol);
      }

      if(num == 0) {
        break;
      }
    }
    return roman.toString();
  }
}
```
